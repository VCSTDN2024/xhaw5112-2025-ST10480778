} else if (courseCount === 2) { discountRate = 0.05; } let discount = subtotal * discountRate; let vat = 0.15 * (subtotal - discount); let total = subtotal - discount + vat; localStorage.setItem("courseDetails", JSON.stringify(courseDetails));
